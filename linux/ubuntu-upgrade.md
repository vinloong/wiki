```shell

# 更新 apt 列表
sudo apt-get update 

# 升级软件
sudo apt-get upgrade 

# 移除无用软件
sudo apt-get autoremove 

sudo apt dist-upgrade

# reboot

apt-get install update-manager-core

# do-release-upgrade

# 服务器版
do-release-upgrade -m server

# 桌面版本
# do-release-upgrade -m desktop

# 升级尽可能使用官方源，以避免一些不必要的问题

```
